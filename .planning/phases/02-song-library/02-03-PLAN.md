    export interface ParsedSongPreview {
      title: string
      ccliNumber: string
      author: string
      themes: string[]
      notes: string
      vwType: null  // always null on import, user categorizes later
      teamTags: string[]
      arrangements: Arrangement[]

    }

    Column mapping strategy (defensive, per RESEARCH.md pitfall):
    - Title: `row['Title'] ?? row['Song Title'] ?? ''`
    - CCLI: `row['CCLI Number'] ?? row['CCLI'] ?? row['CCLI #'] ?? ''`
    - Author: `row['Author'] ?? row['Copyright'] ?? ''`
    - Themes: `row['Themes'] ?? row['Tags'] ?? ''` (split on comma)
    - Notes: `row['Notes'] ?? ''`
    - Arrangement N Name: `row['Arrangement N Name'] ?? row['Arrangement N'] ?? ''`
    - Arrangement N BPM: parse as number or null
    - Arrangement N Keys: `row['Arrangement N Keys'] ?? row['Arrangement N Key'] ?? ''`
    - Arrangement N Length: parse as seconds or null
    - Arrangement N Tags: split on comma for teamTags

    Each arrangement gets `id: crypto.randomUUID()`. Song-level teamTags = `[...new Set(arrangements.flatMap(a => a.teamTags))]`.
